Create a pipe for sending and receiving data

Example

Demonstrate pipe construction, and how to send and receive data. For a larger example, see the module-level documentation.

use imxrt_async_hal as hal;
use hal::dma;
use hal::ral::{self, dma0, dmamux, ccm};

let ccm = ral::ccm::CCM::take().unwrap();
// DMA clock gate on
ral::modify_reg!(ral::ccm, ccm, CCGR5, CG3: 0b11);

let dma = dma0::DMA0::take().unwrap();
let mut channels = dma::channels(
    dma,
    dmamux::DMAMUX::take().unwrap(),
);
let (mut tx, mut rx) = dma::pipe::new(channels[29].take().unwrap());

// In the sending task
tx.send(&5i32).await.unwrap();

// In the receiving task
assert_eq!(rx.receive().await.unwrap(), 5i32);
